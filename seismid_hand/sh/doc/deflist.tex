{\begin{list}{}%                            3
{\settowidth{\labelwidth}{\bf #1}%          4
\setlength{\leftmargin}{\labelwidth}%       5
\addtolength{\leftmargin}{\labelsep}%       6
\renewcommand{\makelabel}{\deflabel}}}%     7
{\end{list}}%                               8
